## 🔧 自動化建議

### 憑證監控腳本

```bash
#!/bin/bash
# 保存為: scripts/ssl/check-certificates.sh

aws --profile gemini-pro_ck s3 ls s3://renew-ssl-certification/ | \
awk '{print $2}' | sed 's/\///' | sort | while read domain; do
    aws --profile gemini-pro_ck s3 cp \
        s3://renew-ssl-certification/${domain}/fullchain.pem \
        /tmp/${domain}.pem > /dev/null 2>&1

    if [ -f "/tmp/${domain}.pem" ]; then
        days=$(openssl x509 -in "/tmp/${domain}.pem" -noout -enddate | \
               awk -F= '{print $2}' | xargs -I {} date -j -f "%b %d %H:%M:%S %Y %Z" "{}" "+%s" | \
               awk -v now=$(date +%s) '{print int(($1-now)/86400)}')

        if [ $days -lt 30 ]; then
            echo "⚠️ $domain: $days 天剩餘"
        fi
    fi
done
```

### 設定 cron 定期檢查

```bash
# 每週一早上 9:00 檢查
0 9 * * 1 /path/to/scripts/ssl/check-certificates.sh | mail -s "SSL Certificate Status" admin@example.com
```
